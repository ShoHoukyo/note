type vs. interface​

https://github.com/microsoft/TypeScript/issues/15300

interface IndexType {
  [key: string]: string;
}

interface doesNotWork {
  hola: string;
}
type doWorks = { hola: string };

let varIndexType: IndexType;
varIndexType = { hola: 'hello' };

const correct: doWorks = { hola: 'hello' };
varIndexType = correct;

const itypeConst: doesNotWork = { hola: 'hello ' };
varIndexType = itypeConst; //Index signature is missing in type 'doesNotWork'
varIndexType = { ...itypeConst }; //workaround but not equivalent since the instance is not the same
varIndexType = itypeConst as unknown as IndexType; //workaround
