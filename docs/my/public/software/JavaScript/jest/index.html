jest

jest.spyOn(module.testables, 'functionName') doesn't throw an error, but it doesn't allow for proper mocking.

surprise​

Jest ("29.5.0" -> 29.7), restoreAllMocks no longer also calls clearAllMocks

mock deep​

https://jestjs.io/docs/jest-object

Modules that are mocked with jest.mock are mocked only for the file
that calls jest.mock. Another file that imports the module will get
the original implementation even if it runs after the test file that
mocks the module.


// spyOn works, watch out for jest.resetAllMocks();
jest.spyOn(module, 'func')

// always returns a simple string, can take unknown types and strange objects as input
function stringifyError(err) {
  // need to rm 'stack'
  // JSON.stringify not work with circular reference, may need https://github.com/douglascrockford/JSON-js/blob/master/cycle.js
  if (err instanceof Object) return JSON.stringify(err, Object.getOwnPropertyNames(err));
  else return err.toString();
  //   else return `${err}`;
}

try {
    func() // expect failure
    fail("Shouldn't reach this point");
    // OR  throw new Error("Shouldn't reach this point");
} catch (error) {
    expect(error).toMatchInlineSnapshot(`[abcError]`);
    
    // below not good
    expect(stringifyError(error)).
    expect(
            JSON.stringify(error, 
            Object.getOwnPropertyNames(error)  // need to remove 'stack' though as it's unlikely to match
            )
          ).toMatchInlineSnapshot();
    expect({
            name: error.name,
            message: error.message,
            cause: error.cause,
          }).toMatchSnapshot();
          
          

// https://github.com/facebook/jest/blob/bc26cd79e60b7bf29d854293f0f011936fda1a5a/website/versioned_docs/version-29.5/MockFunctions.md#L179

jest.mock('./path');
import {bar, foo} from './path';
const fooMock = foo as jest.Mock;  // best compromise?


jest.mock('../foo', () => {
  const originalModule = jest.requireActual('../foo');

  const barMock = jest.fn();  // not accessible outside
  return {
    __esModule: true,
    ...originalModule,
    bar: barMock,   // without 'import' statement, may still work?
    default: jest.fn(() => 'mocked baz'),
    foo: 'mocked foo',  // to mock foo only, no need for 'import'
    
    exportToMock: {
        functionInExportToMock: ()=>({
            fieldReturnedByFunction: jest.fn();
        })
    }
    
  };
});

import {bar, foo} from './path';
// still needed, p3 try dynamic import
// bar, foo need to be referenced   TODO big problem

const fooMock = jest.fn();
mockModule({
  modulePath: './path',
  mockMap: {
    foo: fooMock
  },
})

function mockModule({
  modulePath,
  mockMap,
}: {
  /**
   * e.g. './path' as in 'import x from "./path"'
   */
  modulePath: string;

  /**
   * mock mapping  e.g. {foo: fooMock}
   */
  mockMap: Record<string, unknown>;
}) {
  jest.mock(modulePath, () => {
    const originalModule = jest.requireActual(modulePath);

    return {
      __esModule: true,
      ...originalModule,
      ...mockMap,
    };
  });
}
