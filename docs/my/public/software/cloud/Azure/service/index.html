name: Azure App Service
overview: HTTP-based service for hosting web applications, REST APIs, and mobile back ends
features:
  - "Scale out/in (machines, by autoscaling)"
  - "Scale up/down (CPU, RAM)"
  - "CI/CD and manual deploy"
  - "Deployment slot (swap STG PROD)"
  - "various language (Python, etc) runtime"
  - "out-of-the-box authentication (of end-user) with federated identity providers like facebook"
limits:
  - "built-in image higher latency read storage"
references:
  - "https://learn.microsoft.com/en-us/training/modules/introduction-to-azure-app-service/2-azure-app-service"
  - "https://learn.microsoft.com/en-us/training/modules/introduction-to-azure-app-service/3-azure-app-service-plans"
  - "https://learn.microsoft.com/en-us/training/modules/introduction-to-azure-app-service/6-network-features"
ServicePlan:
  overview: "an app always runs in an App Service plan"
  defines:
    - "OS"
    - "Region"
    - "Number, size of VM"
    - "Pricing tier"
  use:
    - "save money by putting apps in same App Service plan"
    - "scale independently by using separate App Service plan"
  deploymentSlot:
    use:
      - "stg, prod swap"
      - "percentage testing: random assign client sessions"
  autoscaling:
    DenialOfService:
      overview: "better filter before reaching service (autoscaling would be expensive)"
    use:
      - "overhead monitoring resources and trigger scaling event, in long-term better manually scale system"
      - "https://learn.microsoft.com/en-us/training/modules/scale-apps-app-service/3-app-service-autoscale-conditions-rules"
network:
  overview: "by default, only internet"
deploy:
  command: "az webapp up"
  exampleResult: {"URL": "http://az204app10680.azurewebsites.net", "appserviceplan": "asp_3", "location": "eastus", "name": "az204app10680", "os": "Windows", "resourcegroup": "rg"}
  
webapp:
  config:
    overview: "app settings passed as environment variables (encrypted-at-rest)"
    ConnectionString:
      overview: "similar to app-setting"
    GeneralSetting:
      - "StackSetting (think Docker Image)"
      - "PlatformSetting"
      - debugging
      - clientCert
    pathMapping:
      overview: "think Docker Container volume"
    logging:
      overview: "app, deployment logging"
      references:
        - "https://learn.microsoft.com/en-us/training/modules/configure-web-app-settings/5-enable-diagnostic-logging"
